⚙️ Configurar Base de Datos
Conexión a Supabase (PostgreSQL)
🔗 Credenciales Supabase
1. Ve a supabase.com → tu proyecto → Project Settings → API
2. Copia Project URL y anon public key
3. Ejecuta el SQL de abajo en SQL Editor de Supabase
4. Ingresa las credenciales y haz clic en "Conectar y Guardar"
📋 Script SQL — ejecutar en Supabase → SQL Editor
CREATE TABLE IF NOT EXISTS proyectos (id BIGSERIAL PRIMARY KEY,fr TEXT,nombre TEXT,inicio DATE,fin DATE,real INTEGER DEFAULT 0,complejidad TEXT,flujo TEXT,jp TEXT,area TEXT,descripcion TEXT,presupuesto BIGINT DEFAULT 0,tiene_ia BOOLEAN,es_estrategico BOOLEAN,consume_nube BOOLEAN,es_pci BOOLEAN,created_at TIMESTAMPTZ DEFAULT NOW(),updated_at TIMESTAMPTZ DEFAULT NOW());
CREATE TABLE IF NOT EXISTS tareas (id BIGSERIAL PRIMARY KEY,proyecto_id BIGINT REFERENCES proyectos(id) ON DELETE CASCADE,nombre TEXT,inicio DATE,fin DATE,pct INTEGER DEFAULT 0,responsable TEXT,parent_id BIGINT,orden INTEGER DEFAULT 0,created_at TIMESTAMPTZ DEFAULT NOW());
CREATE TABLE IF NOT EXISTS informes (id BIGSERIAL PRIMARY KEY,proyecto_id BIGINT REFERENCES proyectos(id) ON DELETE CASCADE,responsable TEXT,fecha DATE,pct INTEGER DEFAULT 0,estado TEXT,detalle TEXT,proximos TEXT,obstaculos TEXT,created_at TIMESTAMPTZ DEFAULT NOW());
CREATE TABLE IF NOT EXISTS riesgos (id BIGSERIAL PRIMARY KEY,proyecto_id BIGINT REFERENCES proyectos(id) ON DELETE CASCADE,nivel TEXT,descripcion TEXT,responsable TEXT,created_at TIMESTAMPTZ DEFAULT NOW());
CREATE TABLE IF NOT EXISTS documentos (id BIGSERIAL PRIMARY KEY,proyecto_id BIGINT REFERENCES proyectos(id) ON DELETE CASCADE,nombre TEXT,tipo TEXT DEFAULT '📄',tamano TEXT,url TEXT,created_at TIMESTAMPTZ DEFAULT NOW());
CREATE TABLE IF NOT EXISTS comentarios (id BIGSERIAL PRIMARY KEY,proyecto_id BIGINT REFERENCES proyectos(id) ON DELETE CASCADE,autor TEXT,texto TEXT,created_at TIMESTAMPTZ DEFAULT NOW());
ALTER TABLE proyectos ENABLE ROW LEVEL SECURITY;ALTER TABLE tareas ENABLE ROW LEVEL SECURITY;ALTER TABLE informes ENABLE ROW LEVEL SECURITY;ALTER TABLE riesgos ENABLE ROW LEVEL SECURITY;ALTER TABLE documentos ENABLE ROW LEVEL SECURITY;ALTER TABLE comentarios ENABLE ROW LEVEL SECURITY;
CREATE POLICY "ap" ON proyectos FOR ALL USING(true)WITH CHECK(true);CREATE POLICY "at" ON tareas FOR ALL USING(true)WITH CHECK(true);CREATE POLICY "ai" ON informes FOR ALL USING(true)WITH CHECK(true);CREATE POLICY "ar" ON riesgos FOR ALL USING(true)WITH CHECK(true);CREATE POLICY "ad" ON documentos FOR ALL USING(true)WITH CHECK(true);CREATE POLICY "ac" ON comentarios FOR ALL USING(true)WITH CHECK(true);
